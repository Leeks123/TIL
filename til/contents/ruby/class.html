# Class

class NewClass
  # instance var
  @name
  # global var
  $files
  # class var
  @@files
  
  # 초기화
  def initialize(name)
    @name = name
  end
end

# 인스턴스 생성

firstObject = NewClass.new("tom")

# 상속

class parentClass
	def greeting
    puts "Hello!"
  end
end

class childClass < parentClass
end
  
child = childClass.new
child.greeting
# ==> Hello!

# override

class childClass < parentClass
  def greeting
    super
    puts "Bye!"
  end
end

루비는 다중상속을 지원하지 않는다. 그러나 mixin 을 이용해 비슷하게 구현가능하다

# Public / Private

class ClassName
  # Some class stuff

  public
  # Public methods go here
  def public_method; end

  private
  # Private methods go here
  def private_method; end
end

# attr_reader, attr_writer

class Person
  attr_reader :name
  attr_writer :name
  def initialize(name)
    @name = name
  end
end

위와 같이 지정해놓으면 루비는 다음과 같이 해석한다

def name
  @name
end

def name=(value)
  @name = value
end

만약 read, write를 동시에 하는 변수라면 attr_accssor 로 지정하면 됨
