# Loop & Iterators

# while

i = 0
while i<5
  puts i
  i = i+1
end

# until loop

i = 0
until i == 6
  i += 1
end

# for

for num in 1..5
  print num,' '
end
# ==> 1 2 3 4 5

for num in 1...5
  print num,' '
end
# ==> 1 2 3 4

a...b 은 a 부터 b-1까지, a..b 는 a 부터 b까지

# loop Method

loop { print 'hello' } # infinite

위와 같은 방법으로 iterator를 생성할 수 있다. { } 는 do 와 end 로 대체할 수 있다

i = 0
loop do
  i += 1
  next if i%2 == 0
  print "#{i}"
  break if i > 10
end

next 는 다른언어에서의 continue. 조건이 맞다면 스킵한다.

# .each

arr = [1,2,3,4,5]

arr.each {|item| print item*2,' '}
# ==> 2 4 6 8 10

# .times

10.times{ print 'a'}
# ==> aaaaaaaaaa
