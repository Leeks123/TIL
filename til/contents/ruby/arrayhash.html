# hash

hash = {
  key1 => value1,
  key2 => value2,
  key3 => value3
}

위처럼 직접 생성할 수도 있지만 다음과 같이도 가능

hash = Hash.new

hash["key1"] = "value1"

# hash sort

sorted = hash.sort_by {|k,v| v} # v(value)를 기준으로 정렬
