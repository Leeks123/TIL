# vector

# 생성

vector<int> v1;	// 빈 벡터 생성
vector<int> v2(5); // 사이즈가 5인 벡터 생성, 전부 0
vector<int> v3(5,1);	// 사이즈가 5이고 전부 1
vector<int> v4(v2);	// v2를 복사해서 v4 생성

# 추가 / 삭제

v1.push_back(1);	// 벡터 뒤에 1을 추가
v1.pop_back();	// 벡터 맨 뒤 원소 삭제

v1.clear();	// 모든 원소 삭제. 원소만 제거하고 메모리는 남아있음

v1.erase(vec1.begin(), vec1.end()); //[a, b) 주소 구간에 해당하는 원소 삭제
v1.erase(iter);	// iter가 가리키는 원소 삭제
  
v1.insert(n,m); // n번쨰 자리에 m값을 삽입
v1.insert(a,b,c); // a번째 자리에 b개의 c값을 삽입

# 할당 / 사이즈 관련

v1.assign(m,n);	// m 사이즈의 벡터 할당, 값은 전부 n
  
v1.size();	// 벡터 원소 갯수 반환
v1.capacity();	// 벡터가 할당된 크기 반환, size랑은 다름
  
v1.resize(n)	// 벡터 n으로 크기 재설정
  
v1.empty();	// 비었으면 true 아니면 false

# 참조

v1.at(idx);	// idx번째 원소 참조. 범위 점검하기 때문에 안전
v1[idx];	// idx번째 원소 참조. 범위 점검 안하기 때문에 빠름
  
v1.front();	// 첫번째 원소 참조
v1.back();	// 마지막 원소 참조
  
v1.begin();	// 벡터의 첫 원소의 주소
v1.end();	// 마지막 원소의 다음 주소 리턴
