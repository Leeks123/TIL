# 정규표현식

# 생성

var pattern = /test/;

var pattern = new RegExp("test");

# 플래그

  • i : 정규표현식이 대소문자를 구별하지 않도록
    /test/i => "TEST","Test","tEsT" 등이 매치
  • g : 모든 부분을 매치
  • m : 여러 줄을 포함하는 문자열을 대상으로 매치

# 용어와 연산자

# 정확히 같은 부분 매칭

var pattern = /test/;

위 정규표현식에는 용어(term)이 네 개가 존재하고, 이 패턴에 매치되는 문자열은 문자열 내에 해당 문자들이 그대로 포함되어 있어야 함

# 문자 클래스를 이용해서 매칭하기

var pattern1 = /[abc]/;
var pattern2 = /[^abc]/;
var pattern3 = /[a-m]/;
  • pattern1의 경우 "a","b","c" 문자 중에서 하나를 매치하고자 함
  • pattern2의 경우 캐럿(^) 문자를 통해서 "a","b","c" 문자를 제외한 모든문자를 매치하고자 함
  • pattern3 는 [abcdefghijklm]과 같은 의미

# 이스케이핑

[ , $ , ^ 등의 특수문자를 문자 그 자체로 처리하려면 역슬래시를 이용한다.

# 시작과 끝

  • 캐럿(^)문자가 첫번째 문자로 사용될 경우, 문자열의 시작과 매치
  • $ 표시는 지정한 패턴이 문자열의 끝에 와야 함을 의미함

# 반복

  • ? : 있거나 없거나

    var pattern = /t?est/; // "test", "est"

  • + : 한 번 이상

    var pattern = /t+est/; // "test", "ttest", "tttest" ...

  • * : 0번 이상

    var pattern = /t*est/; // "est", "test", ...

  • {n} : n번 반복

    var pattern = /a{4}/; // "aaaa"

  • {n,m} : n~m번 반복

    var pattern = /a{4,10}/; // "aaaa", "aaaaaa", "aaaaaaaa" ...

# 미리 정의된 문자 클래스

미리 정의된 용어 대상
\t 수평 탭
\b 백스페이스
\v 수직 탭
\f 폼피드
\r 개행 문자
\n 줄바꿈 문자
\cA : \cZ 제어 문자
\u0000 : \uFFFF 유니코드 16진수
\x00 : \xFF 아스키 16진수
. 줄바꿈 문자를 제외한 모든 문자
\d 0~9 사이의 10진수. [0-9]와 같음
\D 0~9 사이의 10진수 숫자에 속하지 않는 모든 문자

\s 공백문자
\S 공백문자 제외한 문자
\b 단어의 경계를 나타내는 문자
\B 단어 내에서 문자의 경계가 아닌 문자

# 매치된 부분을 캡처하기

# 전역 정규표현식

var html = "<div class='test'><b>Hello</b><i>world</i></div>";

var result = html.match(/<(\/?)(\w+)([^>]*?)>/);
console.log(result)

var all = html.match(/<(\/?)(\w+)([^>]*?)>/g);
console.log(all)

(4) ["<div class='test'>", "", "div", " class='test'"]

(6) ["<div class='test'>", "<b>", "</b>", "<i>", "</i>", "</div>"]

전역플래그가 없는 지역 정규 표현식을 String 객체의 match() 메서드와 함께 사용하면, 일치한 문자열 전체와 캡처된 부분들이 담긴 배열이 반환됨.


전역 정규 표현식을 이용하면 첫번째 매치된 문자열만 찾지 않고 모든 매치되는 문자열을 찾아냄. 각 매치 결과 내에 포함된 캡처는 반환하지 않음

# 캡처 참조

var html = "<div class='test'><b>Hello</b><i>world</i></div>";

var pattern = /<(\w+)([^>]*)>(.*?)<\/\1>/g;
var match = pattern.exec(html);
console.log(match)

[
	"<div class='test'><b>Hello</b><i>world</i></div>", 
	"div", 
	" class='test'", 
	"<b>Hello</b><i>world</i>"
]

전체 태그, 태그명, 태그속성, 태그 내용이 순서대로 반환된다. \1 을 이용해서 정규 포현식 내의 첫번째 캡처를 참조한다.

# 함수를 이용해서 치환하기

String 객체의 replace 메서드는 첫번째 인자로 정규표현식, 두번째 인자로 고정 문자열 혹은 함수를 인자로 사용할 수 있다.

function upper(all,letter,i,str) {
  console.log(all, letter, i, str);
  return letter.toUpperCase();
}

console.log("border-bottom-width".replace(/-(\w)/g,upper));

-b b 6 border-bottom-width
-w w 13 border-bottom-width
borderBottomWidth

두번쨰 인자로 함수가 넘겨질 경우, 함수는 다음과 같은 매개변수를 기지고 호출된다

  • 매치되는 전체 문자열
  • 해당 매치 내에 포함된 캡처들
  • 원본 문자열에서 매치되는 부분의 위치
  • 원본 문자열

# 정규표현식 활용

# 문자열 공백 제거

function trim(str){
  return (str || "").replace(/^\s+|\s+$/g,""); // 시작이나 끝에 위치한 공백문자를 ""로 대체해서 제거
}

# 개행문자 찾기

var html = "<b>hello</b>\n<i>world!</i>";

console.log(/.*/.exec(html)[0]);

console.log(/[\S\s]*/.exec(html)[0]); // 공백문자가 아닌 모든 문자와 공백문자의 합집합

console.log(/(?:.|\s)*/.exec(html)[0]);

<b>hello</b>
<i>world!</i>

<i>world!</i>
