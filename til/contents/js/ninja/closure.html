# 클로저

# 기본 동작

var outerValue = 'ninja';
var later;

function outerFunction() {
  var innerValue = 'samurai';
  
  function innerFunction(paramValue) {
    console.log("outerValue",outerValue);
    console.log("innerValue",innerValue);
    console.log("paramValue",paramValue);
    console.log("tooLate",tooLate);
  }
  later = innerFunction;
}

var tooLate = 'ronin';
outerFunction();

later('wakizashi');

outerValue ninja
innerValue samurai
paramValue wakizashi
tooLate ronin

클로저는 함수가 선언된 시점의 유효범위에 있는 모든 함수와 변수를 가지고 있으며, 필요 할 때 그것들을 사용할 수 있다.

  • 함수 매개변수는 함수의 클로저에 포함되어 있다.
  • outer 유효 범위에 속한 모든 변수는, 심지어 함수를 선언한 뒤에 선언된 변수일지라도, 모두 클로저에 포함됨
  • 같은 유효 범위에 속한 변수라 하더라도, 선언에 앞서 참조하는 것은 불가능

# 활용

# Private 변수

function Ninja() {
  var feints = 0;
  this.getFeints = function(){
    return feints;
  };
  this.feint = function(){
    feints++;
  }
}

var ninja = new Ninja();
ninja.feint();

console.log(ninja.getFeints()); // 1
console.log(ninja.feints); 			// undefined

함수 내부의 변수를 직접 접근은 불가능하지만, 내부 메서드를 통해 값을 얻어오거나, 조작하는 것은 가능함

# 콜백과 타이머

클로저는 단순히 생성시점에 유효범위의 상태를 순간포착한 것이 아니라, 외부에는 노출하지 않고 유효범위의 상태를 수정할 수 있게 해주는 적극적인 정보 은닉 수단이다.
