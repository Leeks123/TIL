# 클로져

const x = 1;

function outer() {
  const x = 10;
  const inner = function () {
    console.log(x);
  }
  return inner;
}

const innerFunc = outer();
innerFunc(); // 10

외부 함수보다 중첩 함수가 더 오래 유지되는 경우 중첩 함수는 이미 생명 주기가 종료한 외부 함수의 변수를 참조할 수 있다. 이게 바로 클로저

function foo() {
  const x = 1;
  const y = 2;

  function bar() {
    const z = 3;
    debugger;

    console.log(x);
  }
  return bar;
}

const bar = foo();
bar();

위 코드는 중첩함수가 외부 함수의 변수를 참조하지 않는다면 모던브라우저는 최적화를 통해 상위 스코프를 기억하지 않는다. 이런 경우는 보통 클로저가 아니다.

# 활용

클로저는 상태를 은닉하고 특정 함수에게만 상태 변경을 허용하기 위해 사용한다.

const increase = (function() {
  let num = 0;
  return function() {
    return ++num;
  };
}());

console.log(increase()); // 1
console.log(increase()); // 2
console.log(increase()); // 3

# 캡슐화, 정보은닉

function Person(name, age) {
  this.name = name; // public
  let _age = age; 	// private
  
  this.sayHi = function() {
    console.log(`Hello! I'm ${name}. I'm ${_age}`);
  };
}

const me = new Person('Lee',28);
me.sayHi(); // Hello! I'm Lee. I'm 28
console.log(me.name); // Lee
console.log(me._age); // undefined

위 코드는 sayHi 메서드가 인스턴스 메서드이다. 약간 수정을 해보면

const Person = (function Person(name, age) {
  let _age = 0;
  
  function Person(name, age) {
    this.name = name;
    _age = age;
  }
  
  Person.prototype.sayHi = function() {
    console.log(`Hello! I'm ${this.name}. I'm ${_age}`);
  };
  
  return Person;
}());

const me = new Person('Lee',28);
me.sayHi(); // Hello! I'm Lee. I'm 28
console.log(me.name); // Lee
console.log(me._age); // undefined

const you = new Person('Kim',30);
me.sayHi(); // Hello! I'm Kim. I'm 30
console.log(me.name); // Kim
console.log(me._age); // undefined

me.sayHi(); // Hello! I'm Kim. I'm 30

Person.prototype.sayHi 메서드는 즉시 실행 함수가 호출될 때 생성되는 클로저라서 인스턴스들의 상태가 유지되지 않는다. 정보 은닉의 완전한 지원은 Class 로~~
