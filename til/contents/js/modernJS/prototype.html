# 프로토타입

# 상속과 프로토타입

function Circle(radius) {
  this.radius = radius;
  this.getArea = function() {
    return Math.PI = this.radius ** 2;
  };
}

const circle1 = new Circle(1);
const circle2 = new Circle(2);

console.log(circle1 === circle2); // false
console.log(circle1.getArea === circle2.getArea); // false

생성자 함수를 통해 생성하는 인스턴스는 각각이 모두 프로퍼티와 메서드를 가진다. 중복으로 인한 메모리 낭비가 있을 수 있다.

function Circle(radius) {
  this.radius = radius;
}
Circle.prototype.getArea = function() {
    return Math.PI = this.radius ** 2;
};

const circle1 = new Circle(1);
const circle2 = new Circle(2);

console.log(circle1 === circle2); // false
console.log(circle1.getArea === circle2.getArea); // true

자바스크립트는 프로토타입을 기반으로 상속을 구현한다.

# 프로토타입 객체

프로토타입 객체는 어떤 객체의 상위 객체의 역할을 하는 객체

# __proto__ 접근자 프로퍼티

모든 객체는 __proto__ 접근자 프로퍼티를 통해 자신의 프로토타입, [[Prototype]] 내부 슬롯에 간접적으로 접근할 수 있다.
객체가 자신의 프로토타입에 접근 또는 교체하기 위해 사용

# 함수 객체의 prototype 프로퍼티

함수 객체만이 소유하는 prototype 프로퍼티는 생성자 함수가 생성할 인스턴스의 프로토타입을 가리킴.
non-constructor 함수들은 프로토타입을 소유, 생성하지 않는다.

# 프로토타입의 constructor 프로퍼티와 생성자 함수

constructor 프로퍼티는 prototype 프로퍼티로 자신을 참조하고 있는 생성자 함수를 가리킨다.

# 프로토타입의 생성 시점

# 사용자 정의 생성자 함수와 프로토 타입 생성 시점

console.log(Person.prototype); // {constructor: f}

function Person(name) {
  this.name = name;
}

생성자 함수가 생성될 시점에 constructor가 포함된 프로토타입도 생성되어 생성자 함수의 prototype 프로퍼티에 바인딩된다.

# 빌트인 생성자 함수와 프로토타입 생성 시점

모든 빌트인 생성자 함수는 전역 객체가 생성되는 시점에 생성되고, 프로토타입도 이때 생성되어 바인딩된다.

# 프로토타입 체인

자바스크립트는 객체의 프로퍼티에 접근하려고 할 때 해당 객체에 접근하려는 프로퍼티가 없다면 부모 역할을 하는 프로퍼티를 순차적으로 검색한다.

function Super() {}
function Sub() {}
Sub.prototype = new Super();
sub = new Sub();

Object.getPrototypeOf(sub) === Sub.prototype // true
Object.getPrototypeOf(Sub.prototype) === Super.prototype // true
Object.getPrototypeOf(Super.prototype) === Object.prototype // true

# 오버라이딩과 프로퍼티 쉐도잉

function Person(name){
  this.name = name;
}
Person.prototype.sayHello = function() {
  console.log(`Hi! I'm ${this.name}`);
}

const me = new Person('Lee'); 
me.sayHello(); // 'Hi! I'm Lee' -- 1
me.sayHello = function() {
  console.log(`Hey! I'm ${this.name}`);
}
me.sayHello(); // 'Hey! I'm Lee' -- 2
  1. 프로토타입 체인에 따라 프로퍼티를 검색하여 메서드를 호출한다.
  2. 인스턴스에 프로토타입 프로퍼티와 같은 프로퍼티를 추가하면 이때의 인스턴스 메서드는 프로토타입 메서드를 오버라이딩(Overriding)하고, 프로토타입 메서드는 가려진다.(Property Shadowing)

# 정적 프로퍼티 / 메서드

정적 프로퍼티/메서드는 인스턴스를 생성하지 않고도 호출할 수 있는 프로퍼티/메서드를 말한다.

function Person(name){
  this.name = name;
}

Person.staticProp = 'static prop';
Person.staticMethod = function () {
  console.log('staticMethod');
}
const me = new Person('Lee'); 

me.staticProp; // undefined
me.staticMethod; // undefined

Person.staticProp; // "static prop"
Person.staticMethod; // "staticMethod"

인스턴스는 정적 프로퍼티/메서드를 참조/호출할 수 없다.
