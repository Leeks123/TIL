# Math

# 숫자 형식

# 고정 소수점

const x = 19.51;
x.toFixed(3); // "19.510"
x.toFixed(2); // "19.51"
x.toFixed(1); // "19.5"
x.toFixed(0); // "20". 버림이 아니라 반올림

# 지수 표기법

const x = 3800.5;
x.toExponential(4); // "3.8005e+3"
x.toExponential(3); // "3.801e+3"
x.toExponential(2); // "3.80e+3"
x.toExponential(1); // "3.8e+3"
x.toExponential(0); // "4e+3". 반올림 결과

# 고정 전체 자리수

let x = 1000;
x.toPrecision(5); // "1000.0"
x.toPrecision(4); // "1000"
x.toPrecision(3); // "1.00e+3"
x.toPrecision(2); // "1.0e+3"
x.toPrecision(1); // "1e+3"

# 다른 진수

const x = 12;
x.toString(); // "12". 10진수
x.toString(10); // "12". 10진수
x.toString(16); // "c". 16진수
x.toString(8); // "14". 8진수
x.toString(2); // "1100". 2진수

# 상수

Math.E // 자연로그의 밑수 : ~2.718
Math.PI // 원주열 : ~3.14

Math.LN2 // 2의 자연로그 : ~0.693
Math.LN10 // 10의 자연로그 : ~2.303
Math.LOG2E // Math.E의 밑수가 2인 로그 : ~1.433
Math.LOG10E // Math.E의 상용로그 : 0.434

Math.SQRT1_2 // 1/2의 제곱근
Math.SQRT2 // 2의 제곱근

# 대수 함수

# 거듭제곱

Math.pow(2,3); // 8
Math.sqrt(16); // 4. 제곱근
Math.cbrt(27); // 3. 세제곱근
Math.exp(x); // Math.pow(Math.E, x)과 같다
Math.expm1(x); // Math.exp(x)-1과 같다

# 로그함수

Math.log(x); // x의 자연로그
Math.log10(x); // x의 상용로그
Math.log2(x); // x의 로그2
Math.log1p(x); // x+1의 자연로그

# 기타

Math.abs(x); // x의 절댓값
Math.sign(x); // x의 부호. 음수면 -1, 양수면 1, 0이면 0
Math.ceil(x); // x의 올림. 
Math.floor(x); // x의 내림
Math.trunc(x); // x의 버림. 소수점 제거
Math.round(x); // x의 반올림
Math.min(x1,x2...); 
Math.max(x1,x2...);

# 의사 난수 생성

Math.random();  // 0이상 1 미만
x+(y-x)*Math.random(); // x이상 y 미만
m + Math.floor((n-m)*Math.random()); // m 이상 n 미만의 정수

매개변수에 각도를 쓸 수 없으므로 라디안 값으로 바꿔야 함

function deg2rad(d){ return d/180*Math.PI; }
function rad2deg(r){ return r/Math.PI*180; }
