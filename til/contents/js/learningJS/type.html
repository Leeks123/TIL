# Number 객체의 프로퍼티

const small = Number.EPSILON; // 근사치는 2.2e-16
const binInt = Number.MAX_SAFE_INTEGER; // 표현할 수 있는 가장 큰 정수
const max = Number.MAX_VALUE; // 표현할 수 있는 가장 큰 숫자
const minInt = Number.MIN_SAFE_INTEGER; // 표현할 수 있는 가장 작은 정수
const min = Number.MIN_VALUE; // 표현할 수 있는 가장 작은 숫자
const nInf = Number.NEGATIVE_INFINITY;
const nan = Number.NaN;
const inf = Number.POSITIVE_INFINITY;

# 숫자로 바꾸기

숫자로 바꿀 수 없는 문자열에서는 NaN 을 반환

const a = parseInt("10 aaa",10); // 문자는 무시, 10진수 10
const b = parseInt("3a",16); // 58
const c = parseFloat("15.5 ccc"); // 10진수 15.5

날짜 객체는 valueOf() 메서드를 이용. UTC 1970.1.1 자정부터 몇 밀리초 지났는지를 나타내는 숫자
