# Array

# 배열 조작

# 추가, 제거

const arr = [1,2,3,4];

// 배열 뒤에서 추가 제거
arr.push(5); // 배열의 늘어난 길이 5를 반환하고 배열을 수정한다.
arr.pop(); // 제거된 요소 5를 반환하고 배열 수정한다.

// 배열 앞에서 추가 제거
arr.unshift(0); // 배열 앞에 추가하고 늘어난 길이 5 반환
arr.shift(); // 배열 앞 요소 제거하고 제거된 값 반환

# 붙이기

const arr [1,2,3];
arr.concat(4,5,6); // [1,2,3,4,5,6]을 반환. arr은 그대로
arr.concat([4,5,6]); // [1,2,3,4,5,6]을 반환. arr은 그대로
arr.concat([4,5],6); // [1,2,3,4,5,6]을 반환. arr은 그대로
arr.concat([4,[5,6]]); // [1,2,3,4,[5,6]]을 반환. arr은 그대로

# 일부 가져오기

const arr = [1,2,3,4,5,6];
// slice 메서드는 arr을 바꾸지 않는다.
arr.slice(3); // [4,5,6]
arr.slice(2,4); // [3,4] 인덱스2부터 안덱스4 전까지
arr.slice(-2); // [4,5,6]

# 임의의 위치에 추가, 제거

const arr = [1,5,7];
// splice는 원본 배열을 수정한다
// splice(수정을 시작할 인덱스, 제거할 요소 숫자(제거 안하면 0), 추가될 요소들)
arr.splice(1,0,2,3,4); // []. arr은 [1,2,3,4,5,7]
arr.splice(5,0,6); // []. arr은 [1,2,3,4,5,6,7]
arr.splice(1,2); // []. arr은 [1,4,5,6,7]

# 요소 교체

// copyWithin은 배열 요소를 복사해서 다른 위치에 붙여넣고, 기존의 요소를 덮어씀
// copyWithin(붙일 위치, 복사 시작 위치, 복사 종료 위치)
const arr = [1,2,3,4];
arr.copyWithin(1,2); // arr은 [1,3,4,4]
arr.copyWithin(2,0,2); // arr은 [1,3,1,3]

# 특정 값으로 채우기

const arr = new Array(5).fill(1); // arr은 [1,1,1,1,1]
arr.fill("a"); // arr은 ["a","a","a","a","a"]
arr.fill("b",1,3); // arr은 ["a","b","b","a","a"]

# 역순, 정렬

const arr = [1,2,3,4,5];
arr.reverse(); // arr은 [5,4,3,2,1]
arr.sort(); // arr은 [1,2,3,4,5]

sort 메서드는 인자로 정렬함수를 받을 수 있다.

# 배열 검색

# indexOf, lastIndexOf

찾고자 하는 것과 정확히 일치하는(===) 첫 번째 요소의 인덱스 반환. 없으면 -1 반환

const o = { name: "jerry" };
const arr = [1,5,"a",o,true,5,[1,2],"9"];
arr.indexOf(5); // 1
arr.lastIndexOf(5); // 5
arr.indexOf("a"); // 2
arr.lastIndexOf("a"); // 2
arr.indexOf({ name: "jerry" }); // -1
arr.lastIndexOf(o); // 3

# findIndex

const arr = [{ id:5, name: "Judith" },{ id:7, name:"Francis" }];
arr.findIndex(o=> o.id===5); // 0
arr.findIndex(o=> o.name==="Francis"); // 1
arr.findIndex(o=> o===3); // -1
arr.findIndex(o=> o.id===15); // -1

# some

조건에 맞는 요소를 찾으면 즉시 검색을 멈추고 true를 반환, 없으면 false

const arr = [5, 7, 12, 15, 17];
arr.some(x => x%2===0); // true
arr.some(x => Number.isInteger(Math.sqrt(x))); // false

# every

배열의 모든 요소가 조건에 맞아야 true, 아니면 false

const arr = [4,6,16,36];
arr.every(x => x%2===0); // true
arr.every(x => Number.isInteger(Math.sqrt(x))); // false

# reduce

reduce 의 첫번째 매개변수는 accumulator. 두번쨰 매개변수 부터는 현재 배열 요소, 인덱스, 배열 자체

const arr = [5,7,2,4];
const sum = arr.reduce((a,x)=> a+=x);

# 문자열 병합

Array.prototype.join 은 매개변수로 구분자 하나를 받고 요소들을 하나로 합친 문자열을 반환한다. 구분자의 기본값은 쉼표(,)이다. 정의되지 않은 요소, 삭제된 요소, null, undefined는 모두 빈 문자열 취급한다

const arr = [1,null,"hello","world",true,undefined];
delete arr[3];
arr.join(); // "1,,hello,,true"
arr.join(''); // "1hellotrue"
